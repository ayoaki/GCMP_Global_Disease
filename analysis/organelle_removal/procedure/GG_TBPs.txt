#download GG files, generate taxonomy, generate TBP of M/T/S samples

#get OTUs
wget -O input/GG_OTUs.qza "https://www.dropbox.com/s/ce6cj4m24pi7doy/99-otus.qza?dl=0"

#get taxonomy
wget -O input/GG_taxonomy.qza "https://www.dropbox.com/s/gvhled47lwe46eo/99_otus.fasta?dl=0"

#feature classifier with GG files
qiime feature-classifier classify-consensus-vsearch --i-query input/all.seqs.qza --i-reference-reads input/GG_OTUs.qza --i-reference-taxonomy input/GG_taxonomy.qza --p-strand plus --p-threads 4 --o-classification input/GG_classification_taxonomy.qza

#TBP visualizers
qiime taxa barplot --i-table M_ft.qza --i-taxonomy input/GG_classification_taxonomy.qza --m-metadata-file input/GCMP_EMP_map_r28.txt --o-visualization output/GG_M_tbp.qzv
qiime taxa barplot --i-table T_ft.qza --i-taxonomy input/GG_classification_taxonomy.qza --m-metadata-file input/GCMP_EMP_map_r28.txt --o-visualization output/GG_T_tbp.qzv
qiime taxa barplot --i-table S_ft.qza --i-taxonomy input/GG_classification_taxonomy.qza --m-metadata-file input/GCMP_EMP_map_r28.txt --o-visualization output/GG_S_tbp.qzv
